# SETUP LAB SICUREZZA ======================================

1. Crea la vm con un nome
2. Type: Linux, Subtype: Other Linux
3. Hardware: Base memory: 8gb, Processors: 2
4. Hard Disk: use an existing ... > ~/large/kali-lab da 80gb

5. Settings > network > Adapter2 > host only Adapter
6. Settings > shared folders > aggiungi dal desktop
7. Settings > display > video memory al massimo

8. Fai uno snapshot
9. Accendi la macchina > devices > shared clipboard > bidirectional
10. Esegui lo script ./setup_lab.sh [NUM_LAB]

# PER ABILITARE docker A USARE IL PROXY D'ATENEO ==========

sudo -i 
mkdir -p /etc/systemd/system/docker.service.d

# successivamente creare il seguente file dove inserire il settaggio del proxy

nano /etc/systemd/system/docker.service.d/http-proxy.conf

# aggiungere al file le seguenti linee  [ Valido per il LAB3 ]

[Service]      
Environment="HTTP_PROXY=http://192.168.123.249:8080"
Environment="HTTPS_PROXY=http://192.168.123.249:8080"

# salvare il file e a questo punto lanciare nell'ordine

systemctl daemon-reload
systemctl restart docker

# se il processo è andato correttamente a buon fine il seguente comando 

systemctl show --property=Environment docker

# dovrebbe restituire in output

Environment=HTTP_PROXY=http://192.168.123.249:8080 HTTPS_PROXY=http://192.168.123.249:8080

# NOTA1: su alcune piattaforme, tra cui la Kali che utilizziamo, i default di docker sono troppo stringenti per alcune app.

echo 'DOCKER_OPTS="--default-ulimit nofile=65535"' >> /etc/default/docker
systemctl restart docker
